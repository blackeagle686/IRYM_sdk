{% if LANGUAGE_CODE == 'ar' %}كيفية استخدام RAG Pipeline{% else %}Using the RAG Pipeline{% endif %}
from phoenix.rag import RAGPipeline

# Initialize the pipeline
pipeline = RAGPipeline()

# Ingest documents automatically (PDF, CSV, TXT, HTML)
await pipeline.ingest_document("data/knowledge_base.pdf")

# Query with semantic search and generation
response = await pipeline.query("What are the key policies in chapter 4?")
print(response.answer)
print(response.sources)

# Instantiates ChromaDB or Qdrant based on config

# Semantic Search
results = await db.search("Speed of Phoenix AI", top_k=1)
print(results[0].text) # Output: "Phoenix AI is fast"

{% if LANGUAGE_CODE == 'ar' %}كيفية استخدام Semantic Cache{% else %}Using Semantic Cache{% endif %}
from phoenix.cache.semantic import SemanticCache

cache = SemanticCache(threshold=0.85) # 85% similarity threshold

# Look up an answer before calling the LLM
cached_answer = await cache.get("How do I install Phoenix?")

if not cached_answer:
    # Expensive LLM Call
    answer = await llm.generate("How to install Phoenix AI SDK?")
    # Cache it for future similar questions!
    await cache.set("How do I install Phoenix?", answer)

from phoenix.core import Application

app = Application()

@app.on_startup
async def init_databases():
    await db.connect()
    print("Databases warmed up!")

@app.on_shutdown
async def cleanup():
    await db.close()
    print("Clean shutdown completed.")

# Run the app (usually handled by FastAPI/ASGI)
await app.start()

{% if LANGUAGE_CODE == 'ar' %}كيفية بناء الوكلاء المستقلين{% else %}Building Autonomous Agents{% endif %}
from phoenix.agent import Agent
from phoenix.tools import tool

# 1. Define a tool
@tool(name="calculator", description="Solves math")
def calc(equation: str):
    return eval(equation)

# 2. Spin up the agent
agent = Agent()
agent.register_tool(calc)

# 3. Watch it think, plan, and execute
result = await agent.run("What is (500 * 2) + 12?", mode="plan")
print(result) # Output: 1012
