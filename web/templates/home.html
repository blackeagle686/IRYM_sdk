from IRYM_sdk import init_irym, startup_irym, get_rag_pipeline

# FastAPI
@app.on_event("startup")
async def startup():
    init_irym()
    await startup_irym()

@app.get("/ask")
async def ask(q: str):
    rag = get_rag_pipeline()
    return {"answer": await rag.query(q)}
