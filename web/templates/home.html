import asyncio
from phoenix import init_phoenix, startup_phoenix, get_rag_pipeline

async def rag_demo():
    init_phoenix()
    await startup_phoenix()
    rag = get_rag_pipeline()

    # 1. Ingest documents (Docs + Source Code .py, .js, etc.)
    await rag.ingest("./my_project")

    # 2. Ingest from GitHub Repository (Automated indexing)
    await rag.ingest_github("https://github.com/blackeagle686/phoenix-ai.git")

    # 3. Query with automatic Citations
    answer = await rag.query("How do I extend the cache layer?")
    print(f"AI Answer: {answer}")

agent_loop.py Autonomous Mode 
from phoenix.agent import Agent
from phoenix.llm.openai import OpenAILLM
from phoenix.memory.hybrid import HybridMemory
from phoenix.tools.registry import ToolRegistry

# Initialize Agent with default components
agent = Agent(
    llm=OpenAILLM(), 
    memory=HybridMemory(), 
    tools=ToolRegistry.load_default()
)

# Run an autonomous Think-Plan-Act task
result = await agent.run(
    "Analyze the latest AI news and save it to news.txt", 
    mode="auto"
)
print(f"Agent Output: {result}")
